using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.Events;

public class AbilityManager : MonoBehaviour
{
	[SerializeField] List<AbilityEvent> _abilityEventList = new List<AbilityEvent>();

	[SerializeField] GameObject _buttonPrefab;

	private void Awake()
	{
		InitializeButtons();
	}

	void InitializeButtons()
	{
		int offset = 0;

		foreach (AbilityEvent aEvent in _abilityEventList)
		{
			if(aEvent != null)
			{
				GameObject abilityButton = Instantiate(_buttonPrefab, GameObject.Find("Canvas").transform);

				abilityButton.GetComponent<RectTransform>().localPosition = new Vector3(0 + offset, -350, 0);
				offset += 160;

				abilityButton.GetComponent<Button>().onClick.AddListener(aEvent.Invoke);
				abilityButton.transform.Find("Text").GetComponent<TMPro.TMP_Text>().text = aEvent.aName;
			}
		}
	}
}
