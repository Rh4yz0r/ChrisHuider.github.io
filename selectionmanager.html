using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Events;

public class SelectionManager : MonoBehaviour
{
	public Unit unitManager;

	public UnityEvent onSelected;
	public UnityEvent onDeselected;

	public bool selected = false;


	private void Start()
	{
		onSelected.AddListener(SelectUnit);
		
		onDeselected.AddListener(DeselectUnit);
	}

	public virtual void SelectUnit()
	{
		selected = true;

		foreach (GameObject member in unitManager.members)
		{
			member.GetComponent<selection_component>().Select();
		}
		
		unitManager.mover.AddMoveEvents();
		unitManager.abilities.ShowButtons();
	}

	public virtual void DeselectUnit()
	{
		selected = false;

		foreach (GameObject member in unitManager.members)
		{
			member.GetComponent<selection_component>().Deselect();
		}
		
		unitManager.mover.RemoveMoveEvents();
		unitManager.abilities.RemoveButtons();
	}
}
