using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.AI;

public class Unit : MonoBehaviour
{
	[Header("Type of Unit")]
	public UnitType unitType;

	[Space]
	public List<GameObject> members = new List<GameObject>();

	[Space]
	public bool DebugUnit = false;


	GameObject _memberPrefab;

	int _maxMembers;
	int _minMembers;

	int _squadLeaderID = 0;
	int _currentMemberCount = 1;

	int _spawnMemberQueue = 0;
	bool spawnAvailable = false;

	bool _selected = false;

	void Awake()
	{
		_currentMemberCount = members.Count;

		InitializeUnitType();
	}


	void Update()
	{
		if (_spawnMemberQueue > 0 && spawnAvailable) //Spawn player and wait for it to finish
		{
			StartCoroutine(SpawnMember());
			spawnAvailable = false;
		}

		if (_currentMemberCount < _minMembers) KillUnit(); //Not enough members to function, Unit is killed
	}

	/// <summary>
	/// Adds a new member to the unit (unless it's full).
	/// </summary>
	/// <returns></returns>
	public IEnumerator SpawnMember()
	{
		if (_currentMemberCount < _maxMembers)
		{
			if(DebugUnit) Debug.Log("Spawning new member");

			_spawnMemberQueue--;
			_currentMemberCount++;

			yield return new WaitForSecondsRealtime(5);

			GameObject newMember;
			newMember = Instantiate(_memberPrefab, this.gameObject.transform);
			members.Add(newMember);
			newMember.transform.position = members[_squadLeaderID].transform.position - (Vector3.right * _currentMemberCount);
			newMember.GetComponent<SquadMember>().manager = this;

			spawnAvailable = true;
		}
		else if (DebugUnit) Debug.Log("Squad full");
	}

	/// <summary>
	/// Removes the last character in the members list.
	/// </summary>
	void KillMember()
	{
		Destroy(members[members.Count - 1]);
		members.Remove(members[members.Count - 1]);
		_currentMemberCount--;

		if (DebugUnit) Debug.Log("Removed member");
	}

	/// <summary>
	/// Removes every character in members list, in turn removing the unit.
	/// </summary>
	/// <param name="dropWeapon"></param>
	void KillUnit(bool dropWeapon = false)
	{
		if (DebugUnit) Debug.Log("Killed unit");

		foreach (GameObject member in members)
		{
			KillMember();
		}

		if (dropWeapon)
		{
			//Drop abandoned weapon
		}
	}

	/// <summary>
	/// Sets every character in the unit's NavMeshAgent destination to MousePosition's positions.
	/// </summary>
	protected virtual void MoveToMousePos()
	{
		MousePosition mousePos = GameObject.Find("MousePosition").GetComponent<MousePosition>();

		int positionID = 0;

		foreach (GameObject member in members)
		{
			member.GetComponent<NavMeshAgent>().SetDestination(mousePos.positions[positionID].position);
			positionID++;
		}
	}

	/// <summary>
	/// Enable selection circle for every member.
	/// </summary>
	public void SelectUnit()
	{
		_selected = true;

		foreach (GameObject member in members)
		{
			member.GetComponent<selection_component>().Select();
		}
	}

	/// <summary>
	/// Disable selection circle for every member.
	/// </summary>
	public void DeselectUnit()
	{
		_selected = false;

		foreach (GameObject member in members)
		{
			member.GetComponent<selection_component>().Deselect();
		}
	}

	/// <summary>
	/// Sets local variables to UnitType values.
	/// </summary>
	void InitializeUnitType()
	{
		_maxMembers = unitType.maxMembers;
		_minMembers = unitType.minMembers;

		_memberPrefab = unitType.memberPrefab;
	}
}
