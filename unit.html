using System.Collections;
using System.Collections.Generic;
using Pathfinding;
using UnityEngine;
using UnityEngine.AI;

public class Unit : MonoBehaviour
{
	[Header("Type of Unit")]
	public UnitType unitType;

	[Space]
	public List<GameObject> members = new List<GameObject>();
	
	[HideInInspector] public int squadLeaderID = 0;
	
	protected GameObject memberPrefab;

	[Space]
	public List<Transform> positions = new List<Transform>();

	protected MousePosition _mousePos;
	
	protected int _maxMembers;
	protected int _minMembers;
	
	protected int _currentMemberCount = 1;

	protected int _spawnMemberQueue = 0;
	protected bool _spawnAvailable = false;

	protected bool _selected = false;

	protected virtual void OnEnable()
	{
		_currentMemberCount = members.Count;

		InitializeUnitType();
		
		_mousePos = GameObject.Find("MousePosition").GetComponent<MousePosition>();
	}
	
	protected virtual void Update()
	{
		if (_spawnMemberQueue > 0 && _spawnAvailable) //Spawn player and wait for it to finish
		{
			StartCoroutine(SpawnMember());
			_spawnAvailable = false;
		}

		if (_currentMemberCount < _minMembers) KillUnit();
	}

	/// <summary>
	/// Adds a new member to the unit (unless it's full).
	/// </summary>
	/// <returns></returns>
	public virtual IEnumerator SpawnMember()
	{
		yield return new WaitForSeconds(0);
	}

	/// <summary>
	/// Removes the last character in the members list.
	/// </summary>
	protected void KillMember()
	{
		Destroy(members[members.Count - 1]);
		members.Remove(members[members.Count - 1]);
		_currentMemberCount--;

		if(DebugUnit) Debug.Log("Removed member");
	}

	/// <summary>
	/// Removes every character in members list, in turn removing the unit.
	/// </summary>
	/// <param name="dropWeapon"></param>
	protected virtual void KillUnit(bool dropWeapon = false) { }

	/// <summary>
	/// Sets every character in the unit's NavMeshAgent destination to MousePosition's positions.
	/// </summary>
	protected virtual void MoveToMousePos() { }

	public virtual void SelectUnit()
	{
		_selected = true;

		foreach (GameObject member in members)
		{
			member.GetComponent<selection_component>().Select();
		}
	}

	public virtual void DeselectUnit()
	{
		_selected = false;

		foreach (GameObject member in members)
		{
			member.GetComponent<selection_component>().Deselect();
		}
	}

	/// <summary>
	/// Sets local variables to UnitType values.
	/// </summary>
	protected void InitializeUnitType()
	{
		_maxMembers = unitType.maxMembers;
		_minMembers = unitType.minMembers;

		memberPrefab = unitType.memberPrefab;
	}
}
