using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Events;

public class Unit : MonoBehaviour
{
	[Header("Type of Unit")]
	public UnitType unitType;

	[Space]
	public List<GameObject> members = new List<GameObject>();
	
	[HideInInspector] public int squadLeaderID = 0; //Temp

	
	//[SerializeField] private bool _selectable;
	[HideInInspector] public SelectionManager selector;

	//[SerializeField] private bool _hasHealth;
	[HideInInspector] public HealthManager health;
	
	//[SerializeField] private bool _canMove;
	[HideInInspector] public MovementManager mover;

	[HideInInspector] public PositionsManager positions;

	[HideInInspector] public AbilityManager abilities;


	protected virtual void OnEnable()
	{
		if (selector == null) SetSelectionManager(this.gameObject);
		if (health == null) SetHealthManager(this.gameObject);
		if (mover == null) SetMovementManager(this.gameObject);
		if (positions == null) SetPositionManager(transform.Find("Positions").gameObject);
		if (abilities == null) SetAbilityManager(this.gameObject);
		
		InitializeUnitType();
	}

	private void SetSelectionManager(GameObject target)
	{
		if (target.GetComponent<SelectionManager>() == null)
		{
			selector = target.AddComponent<SelectionManager>();
			selector.unitManager = this;
		}
		else selector = target.GetComponent<SelectionManager>();
	}

	private void SetHealthManager(GameObject target)
	{
		if (target.GetComponent<HealthManager>() == null)
		{
			health = target.AddComponent<HealthManager>();
			health.unitManager = this;
		}
		else health = target.GetComponent<HealthManager>();
	}

	private void SetMovementManager(GameObject target)
	{
		if (target.GetComponent<MovementManager>() == null)
		{
			mover = target.AddComponent<MovementManager>();
			mover.unitManager = this;
		}
		else mover = target.GetComponent<MovementManager>();
	}

	private void SetPositionManager(GameObject target)
	{
		if (target.GetComponent<PositionsManager>() == null)
		{
			positions = target.AddComponent<PositionsManager>();
			positions.unitManager = this;
		}
		else positions = target.GetComponent<PositionsManager>();
	}
	
	private void SetAbilityManager(GameObject target)
	{
		if (target.GetComponent<AbilityManager>() == null)
		{
			abilities = target.AddComponent<AbilityManager>();
			abilities.unitManager = this;
		}
		else abilities = target.GetComponent<AbilityManager>();
	}

	private void InitializeUnitType()
	{
		health.memberPrefab = unitType.memberPrefab;
		health.maxMembers = unitType.maxMembers;
		health.minMembers = unitType.minMembers;
	}
}
